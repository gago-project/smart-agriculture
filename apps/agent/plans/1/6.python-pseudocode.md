## 2. `FlowState` 伪代码

```python
class SoilAgentState(BaseModel):
    request_id: str
    trace_id: str
    session_id: str
    turn_id: int
    user_input: str
    channel: str = "web"
    timezone: str = "Asia/Shanghai"

    input_type: InputType | None = None
    intent: IntentType | None = None
    answer_type: AnswerType | None = None
    route_target: str | None = None

    raw_slots: SlotBundle = Field(default_factory=SlotBundle)
    merged_slots: SlotBundle = Field(default_factory=SlotBundle)
    context_used: dict = Field(default_factory=dict)

    business_time: BusinessTimeBundle = Field(default_factory=BusinessTimeBundle)
    execution_gate_result: ExecutionGateBundle = Field(default_factory=ExecutionGateBundle)

    query_plan: QueryPlanBundle = Field(default_factory=QueryPlanBundle)
    query_result: QueryResultBundle = Field(default_factory=QueryResultBundle)

    rule_result: RuleResultBundle = Field(default_factory=RuleResultBundle)
    template_result: TemplateResultBundle = Field(default_factory=TemplateResultBundle)
    advice_result: AdviceResultBundle = Field(default_factory=AdviceResultBundle)

    answer_bundle: AnswerBundle = Field(default_factory=AnswerBundle)

    query_log_entries: list[dict] = Field(default_factory=list)
    context_to_save: dict | None = None

    node_trace: list[str] = Field(default_factory=list)
    errors: list[dict] = Field(default_factory=list)
    final_status: str | None = None
    step_count: int = 0
    retry_count: int = 0
    node_visit_counts: dict[str, int] = Field(default_factory=dict)


class NodeResult(BaseModel):
    next_action: str
    state_patch: dict = Field(default_factory=dict)
    node_status: str = "success"
    debug_payload: dict = Field(default_factory=dict)


def push_trace(state: SoilAgentState, message: str) -> None:
    state.node_trace.append(message)


def push_error(state: SoilAgentState, code: str, message: str) -> None:
    state.errors.append({"code": code, "message": message})
```

实现要点：

- 所有节点共享同一个 `SoilAgentState`
- 节点只改自己负责的字段
- `state_patch` 统一通过 merge 规则写回
- 不允许节点随意塞无约束字段

## 3. Flow Runner 伪代码

```python
class SoilChatFlowRunner:
    def __init__(self, nodes: dict[str, FlowNode], routes: RouteRegistry):
        self.nodes = nodes
        self.routes = routes
        self.entrypoint = "input_guard"
        self.max_steps = 32
        self.max_retry_count = 2
        self.terminals = {
            "safe_end",
            "clarify_end",
            "boundary_end",
            "block_end",
            "fallback_end",
            "verified_end",
        }

    async def run(self, state: SoilAgentState) -> SoilAgentState:
        current_node = self.entrypoint

        while current_node:
            state.step_count += 1
            state.node_visit_counts[current_node] = state.node_visit_counts.get(current_node, 0) + 1

            if state.step_count > self.max_steps:
                state.final_status = "fallback_end"
                state.answer_bundle.final_answer = "当前请求执行步骤过多，已安全降级，请缩小问题范围后重试。"
                return state

            node = self.nodes[current_node]
            push_trace(state, f"{current_node}:start")

            result = await node.run(state)
            state = merge_state_patch(state, result.state_patch)
            push_trace(state, f"{current_node}:{result.next_action}")

            next_target = self.routes.resolve(
                node_name=current_node,
                next_action=result.next_action,
                state=state,
            )

            if next_target in self.terminals:
                state.final_status = next_target
                return state

            current_node = next_target

        state.final_status = "fallback_end"
        return state
```

## 4. 节点统一包装伪代码

```python
async def run_node_with_snapshot(
    *,
    state: SoilAgentState,
    node_name: str,
    handler: Callable[[SoilAgentState], Awaitable[NodeResult]],
    debug_service: DebugService,
):
    started_at = now()

    try:
        input_summary = summarize_state_for_node_input(state, node_name)
        push_trace(state, f"{node_name}:start")

        result = await handler(state)
        new_state = merge_state_patch(state, result.state_patch)

        output_summary = summarize_state_for_node_output(new_state, node_name)
        push_trace(new_state, f"{node_name}:{result.next_action}")

        await debug_service.save_node_snapshot(
            trace_id=new_state.trace_id,
            request_id=new_state.request_id,
            session_id=new_state.session_id,
            turn_id=new_state.turn_id,
            node_name=node_name,
            status=result.node_status,
            started_at=started_at,
            finished_at=now(),
            input_summary=input_summary,
            output_summary=output_summary,
        )
        return result

    except Exception as exc:
        push_error(state, "NODE_EXECUTION_FAILED", f"{node_name} failed: {exc}")
        await debug_service.save_node_snapshot(
            trace_id=state.trace_id,
            request_id=state.request_id,
            session_id=state.session_id,
            turn_id=state.turn_id,
            node_name=node_name,
            status="failed",
            started_at=started_at,
            finished_at=now(),
            input_summary=summarize_state_for_node_input(state, node_name),
            output_summary={},
            error_code="NODE_EXECUTION_FAILED",
            error_message=str(exc),
        )
        return NodeResult(next_action="fallback", node_status="failed")
```

```python
async def execution_gate_handler(state: SoilAgentState) -> NodeResult:
    gate_result = execution_gate_service.evaluate(
        intent=state.intent,
        slots=state.merged_slots,
        business_time=state.business_time,
    )

    if gate_result.must_clarify:
        return NodeResult(
            next_action="clarify_end",
            state_patch={
                "execution_gate_result": gate_result,
                "answer_type": AnswerType.CLARIFICATION,
                "answer_bundle": {"final_answer": gate_result.clarify_message},
            },
        )

    if gate_result.blocked:
        return NodeResult(
            next_action="block_end",
            state_patch={
                "execution_gate_result": gate_result,
                "answer_bundle": {"final_answer": gate_result.block_message},
            },
        )

    if gate_result.shrink_applied:
        return NodeResult(
            next_action="shrink_and_continue",
            state_patch={
                "execution_gate_result": gate_result,
                "business_time": gate_result.effective_business_time,
                "merged_slots": gate_result.effective_slots,
            },
        )

    # 当前策略服务默认不会设置 shrink_applied；该分支仅作为预留扩展。

    return NodeResult(
        next_action="continue",
        state_patch={"execution_gate_result": gate_result},
    )
```

```python
async def data_fact_check_handler(state: SoilAgentState) -> NodeResult:
    check_result = fact_check_service.verify(
        answer_type=state.answer_type,
        answer_bundle=state.answer_bundle,
        query_result=state.query_result,
        rule_result=state.rule_result,

            next_action="fallback",
            state_patch={
                "answer_type": "fallback_answer",
```

## 6. Context / Redis 伪代码

```python
class SessionContextRepository:
    async def load_recent_context(self, session_id: str) -> list[dict]:
        raw = await redis_client.get(f"session_ctx:{session_id}")
        if not raw:
            return []
        snapshot = decode_json(raw)
        return snapshot.get("latest_business_turns", [])[-5:]

    async def save_turn_context(self, session_id: str, turn_id: int, turn_context: dict) -> None:
        key = f"session_ctx:{session_id}"
        raw = await redis_client.get(key)
        snapshot = decode_json(raw) if raw else {"latest_business_turns": []}
        turns = snapshot.get("latest_business_turns", [])

        latest_turn_id = max([item.get("turn_id", 0) for item in turns], default=0)
        if turn_id < latest_turn_id:
            return

        filtered = {
            "turn_id": turn_id,
            "created_at": now_iso(),
            "domain": turn_context.get("domain"),
            "region": turn_context.get("region"),
            "time_window": turn_context.get("time_window"),
            "entity_reference": turn_context.get("entity_reference"),
            "last_intent": turn_context.get("last_intent"),
        }

        turns = [item for item in turns if item.get("turn_id") != turn_id]
        turns.append(filtered)
        turns = sorted(turns, key=lambda item: item.get("turn_id", 0))[-5:]

        await redis_client.set(
            key,
            encode_json({"latest_business_turns": turns}),
            ex=3600,
        )
```

上下文示意：

```python
context = {
    "domain": "soil_moisture",
    "region": "南京市",
    "time_window": "最近7天",
    "entity_reference": "玄武区点位",
    "last_intent": "soil_recent_summary",
}
```

约束：

- Redis 不是整份运行态主存储
- 当前方案只保存“可继承的稳定上下文”
- 事实结论与中间查询结果不入 Redis

## 8. FastAPI `/chat` 主入口伪代码

```python
@router.post("/chat", response_model=ChatResponse)
async def chat(
    request: ChatRequest,
    flow_runner = Depends(get_soil_chat_flow_runner),
    query_log_repository = Depends(get_query_log_repository),
    session_context_repository = Depends(get_session_context_repository),

    initial_state = SoilAgentState(

        user_input=request.user_input,

    final_state = await flow_runner.run(initial_state)

    await query_log_repository.insert_many(final_state.query_log_entries)

    if should_save_business_context(final_state):
        await session_context_repository.save_turn_context(
            session_id=final_state.session_id,
            turn_id=final_state.turn_id,
            turn_context=build_turn_context(final_state),
        )

    return ChatResponse(
        request_id=final_state.request_id,
        session_id=final_state.session_id,
        turn_id=final_state.turn_id,
        intent=final_state.intent,
        answer_type=final_state.answer_type,
        final_answer=final_state.answer_bundle.final_answer,
        status=final_state.final_status or "ok",
        trace_id=final_state.trace_id,
    )
```

## 9. 实现注意事项

- 不要让节点自己决定“下一个节点名”，统一由路由层决定
- 不要让模型决定 SQL、规则等级、模板核心槽位
- 不要把完整运行状态塞进 Redis
- 不要把自由规划误当成智能
- 不要把事实题和建议题混在一个生成出口里
